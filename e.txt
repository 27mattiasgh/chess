def draw_arrow(screen, start_pos, end_pos):
    # Calculate the points for the two triangles that make up the arrowhead
    arrow_len = 20
    angle = math.atan2(end_pos[1] - start_pos[1], end_pos[0] - start_pos[0])
    tri1_end = (end_pos[0] - arrow_len * math.cos(angle - math.pi / 6), end_pos[1] - arrow_len * math.sin(angle - math.pi / 6))
    tri2_end = (end_pos[0] - arrow_len * math.cos(angle + math.pi / 6), end_pos[1] - arrow_len * math.sin(angle + math.pi / 6))

    # Draw the line
    pygame.draw.line(screen, ARROW_COLOR, start_pos, end_pos, ARROW_WIDTH)

    # Draw the arrowhead
    pygame.draw.polygon(screen, ARROW_COLOR, [end_pos, tri1_end, tri2_end])
